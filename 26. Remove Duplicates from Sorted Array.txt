// https://leetcode.com/problems/remove-duplicates-from-sorted-array/

class Solution {
    public int removeDuplicates(int[] nums) {
        if (nums.length == 0 ) {
            return 0;
        }
        
        int left = 0;
        int right;
        
        for (right = 1 ; right < nums.length; right++) {
            if ( nums[right] != nums[right-1]) {
                ++left;
                nums[left] = nums[right]; 
            }
        }
        return left+1;
    }
}
